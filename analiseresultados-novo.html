import ast 
def read_single_results(arq):
    with open(arq, 'r') as f:
        retorno = {}
        for line in f:
            a, b = line.split(':')
            retorno[a] = ast.literal_eval(b.strip())
            
        return retorno
    
qpmlist = read_single_results('single_results_qpm.txt')
multiquerylist = read_single_results('single_results_multi.txt')
rfralist = read_single_results('single_results_rfra.txt')

print(len(qpmlist), len(multiquerylist), len(rfralist))

def plot_iterations(arraymedias, titulo):
    n_groups = 10
    plt.rcParams["figure.figsize"] = (14,7)
    fig, ax = plt.subplots()
    index = np.arange(n_groups)
    bar_width = 0.15
    opacity = 0.8

    rects1 = plt.bar(index, arraymedias[:,0], bar_width,
    alpha=opacity,
    color='b',
    label='1')

    rects2 = plt.bar(index + bar_width, arraymedias[:,1], bar_width,
    alpha=opacity,
    color='g',
    label='2')

    rects3 = plt.bar(index + 2*bar_width, arraymedias[:,2], bar_width,
    alpha=opacity,
    color='r',
    label='3')

    rects4 = plt.bar(index + 3*bar_width, arraymedias[:,3], bar_width,
    alpha=opacity,
    color='yellow',
    label='4')

    rects5 = plt.bar(index + 4*bar_width, arraymedias[:,4], bar_width,
    alpha=opacity,
    color='black',
    label='5')

    plt.xlabel('Classe')
    plt.ylabel('Precisao')
    plt.title(titulo)
    plt.xticks(index + bar_width, tuple(x_multiquery))
    plt.legend()

    plt.tight_layout()

    plt.show()

    
plot_iterations(mediaqpm, 'Precisao por iteração QPM')
